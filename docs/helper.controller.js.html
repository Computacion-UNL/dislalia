const bcrypt = require('bcryptjs');
/**
 * @description Metodo para encriptar 
 * @param {*} textPlain 
 * @returns 
 */
const encrypt = async (textPlain) => {
    const hash = await bcrypt.hash(textPlain, 10);
    return hash;
}
